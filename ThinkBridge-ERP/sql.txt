/* ThinkBridge ERP - Full Schema (Current) */

-- =========================
-- 1) CORE / SECURITY
-- =========================
CREATE TABLE dbo.Company (
    CompanyID     INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyName   NVARCHAR(150) NOT NULL,
    Industry      NVARCHAR(100) NULL,
    Status        NVARCHAR(20)  NOT NULL DEFAULT 'Pending',
    CreatedAt     DATETIME2     NOT NULL DEFAULT GETUTCDATE()
);

CREATE TABLE dbo.[User] (
    UserID        INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID     INT NULL,  -- NULL for Super Admin
    Fname         NVARCHAR(150) NOT NULL,
    Lname         NVARCHAR(150) NOT NULL,
    Email         NVARCHAR(150) NOT NULL,
    Password      NVARCHAR(255) NOT NULL,
    Phone         NVARCHAR(30) NULL,
    AvatarUrl     NVARCHAR(255) NULL,
    IsSuperAdmin  BIT NOT NULL DEFAULT 0,
    Status        NVARCHAR(20)  NOT NULL DEFAULT 'Active',
    MustChangePassword BIT NOT NULL DEFAULT 1,
    LastLoginAt   DATETIME2     NULL,
    CreatedAt     DATETIME2     NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_User_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID)
);
CREATE UNIQUE INDEX UX_User_Email ON dbo.[User](Email);

CREATE TABLE dbo.[Role] (
    RoleID        INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    RoleName      NVARCHAR(50)  NOT NULL,
    Description   NVARCHAR(255) NULL
);

CREATE TABLE dbo.UserRole (
    UserRoleID    INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    UserID        INT NOT NULL,
    RoleID        INT NOT NULL,
    AssignedAt    DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_UserRole_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID),
    CONSTRAINT FK_UserRole_Role
        FOREIGN KEY (RoleID) REFERENCES dbo.[Role](RoleID)
);

CREATE TABLE dbo.[Module] (
    ModuleID      INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    ModuleName    NVARCHAR(50)  NOT NULL,
    Description   NVARCHAR(255) NULL
);

CREATE TABLE dbo.Permission (
    PermissionID  INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    ModuleID      INT NOT NULL,
    PermissionKey NVARCHAR(80)  NOT NULL,
    Description   NVARCHAR(255) NULL,
    CONSTRAINT FK_Permission_Module
        FOREIGN KEY (ModuleID) REFERENCES dbo.[Module](ModuleID)
);

CREATE TABLE dbo.RolePermission (
    RolePermissionID INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    RoleID           INT NOT NULL,
    PermissionID     INT NOT NULL,
    CONSTRAINT FK_RolePermission_Role
        FOREIGN KEY (RoleID) REFERENCES dbo.[Role](RoleID),
    CONSTRAINT FK_RolePermission_Permission
        FOREIGN KEY (PermissionID) REFERENCES dbo.Permission(PermissionID)
);

CREATE TABLE dbo.SystemSetting (
    SettingID     INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID     INT NOT NULL,
    SettingKey    NVARCHAR(80)  NOT NULL,
    SettingValue  NVARCHAR(255) NOT NULL,
    UpdatedAt     DATETIME2     NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_SystemSetting_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID)
);

CREATE TABLE dbo.AuditLog (
    LogID         INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID     INT NULL,
    UserID        INT NOT NULL,
    Action        NVARCHAR(120) NOT NULL,
    EntityName    NVARCHAR(60)  NOT NULL,
    EntityID      INT NOT NULL,
    IPAddress     NVARCHAR(45)  NULL,
    CreatedAt     DATETIME2     NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_AuditLog_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID),
    CONSTRAINT FK_AuditLog_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID)
);

-- =========================
-- 2) BILLING
-- =========================
CREATE TABLE dbo.SubscriptionPlan (
    PlanID        INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    PlanName      NVARCHAR(80) NOT NULL,
    Price         DECIMAL(12,2) NOT NULL,
    BillingCycle  NVARCHAR(20) NOT NULL,
    MaxUsers      INT NULL,
    MaxProjects   INT NULL,
    IsActive      BIT NOT NULL DEFAULT 1
);

CREATE TABLE dbo.Subscription (
    SubscriptionID INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID      INT NOT NULL,
    PlanID         INT NOT NULL,
    Status         NVARCHAR(20) NOT NULL DEFAULT 'Trial',
    StartDate      DATE NOT NULL,
    EndDate        DATE NULL,
    CONSTRAINT FK_Subscription_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID),
    CONSTRAINT FK_Subscription_Plan
        FOREIGN KEY (PlanID) REFERENCES dbo.SubscriptionPlan(PlanID)
);

CREATE TABLE dbo.Invoice (
    InvoiceID       INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    SubscriptionID  INT NOT NULL,
    InvoiceNumber   NVARCHAR(50) NOT NULL,
    Amount          DECIMAL(12,2) NOT NULL,
    DueDate         DATE NOT NULL,
    PaidDate        DATE NULL,
    Status          NVARCHAR(20) NOT NULL DEFAULT 'Pending',
    CreatedAt       DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_Invoice_Subscription
        FOREIGN KEY (SubscriptionID) REFERENCES dbo.Subscription(SubscriptionID)
);

CREATE TABLE dbo.PaymentTransaction (
    PaymentID       INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    SubscriptionID  INT NOT NULL,
    InvoiceID       INT NULL,
    Provider        NVARCHAR(40) NOT NULL DEFAULT 'PayMongo',
    ExternalTransactionID NVARCHAR(255) NULL,
    CheckoutSessionID NVARCHAR(255) NULL,
    PaymentMethod   NVARCHAR(40) NULL,
    Amount          DECIMAL(12,2) NOT NULL,
    Currency        NVARCHAR(10) NOT NULL DEFAULT 'PHP',
    Status          NVARCHAR(20) NOT NULL DEFAULT 'Pending',
    PaidAt          DATETIME2 NULL,
    CreatedAt       DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_PaymentTransaction_Subscription
        FOREIGN KEY (SubscriptionID) REFERENCES dbo.Subscription(SubscriptionID),
    CONSTRAINT FK_PaymentTransaction_Invoice
        FOREIGN KEY (InvoiceID) REFERENCES dbo.Invoice(InvoiceID)
);

-- =========================
-- 3) PROJECT MANAGEMENT

CREATE TABLE dbo.Project (
    ProjectID     INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID     INT NOT NULL,
    ProjectCode   NVARCHAR(50) NULL,
    ProjectName   NVARCHAR(150) NOT NULL,
    Description   NVARCHAR(MAX) NULL,
    Status        NVARCHAR(20) NOT NULL DEFAULT 'Planning',
    Progress      DECIMAL(5,2) NULL DEFAULT 0,
    Category      NVARCHAR(50) NULL,
    StartDate     DATE NULL,
    DueDate       DATE NULL,
    CreatedBy     INT NOT NULL,
    CreatedAt     DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_Project_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID),
    CONSTRAINT FK_Project_CreatedBy
        FOREIGN KEY (CreatedBy) REFERENCES dbo.[User](UserID)
);

CREATE TABLE dbo.ProjectMember (
    ProjectMemberID INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    ProjectID       INT NOT NULL,
    UserID          INT NOT NULL,
    MemberRole      NVARCHAR(30) NULL,
    JoinedAt        DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_ProjectMember_Project
        FOREIGN KEY (ProjectID) REFERENCES dbo.Project(ProjectID),
    CONSTRAINT FK_ProjectMember_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID)
);

CREATE TABLE dbo.Task (
    TaskID        INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    ProjectID     INT NOT NULL,
    Title         NVARCHAR(150) NOT NULL,
    Description   NVARCHAR(MAX) NULL,
    Status        NVARCHAR(20)  NOT NULL DEFAULT 'To Do',
    Priority      NVARCHAR(10)  NULL DEFAULT 'Medium',
    StartDate     DATE NULL,
    DueDate       DATE NULL,
    EstimatedHours DECIMAL(8,2) NULL,
    ActualHours   DECIMAL(8,2) NULL,
    CreatedBy     INT NOT NULL,
    CreatedAt     DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_Task_Project
        FOREIGN KEY (ProjectID) REFERENCES dbo.Project(ProjectID),
    CONSTRAINT FK_Task_CreatedBy
        FOREIGN KEY (CreatedBy) REFERENCES dbo.[User](UserID)
);

CREATE TABLE dbo.TaskAssignment (
    TaskAssignmentID INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    TaskID           INT NOT NULL,
    UserID           INT NOT NULL,
    AssignedAt       DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_TaskAssignment_Task
        FOREIGN KEY (TaskID) REFERENCES dbo.Task(TaskID),
    CONSTRAINT FK_TaskAssignment_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID)
);

CREATE TABLE dbo.TaskUpdate (
    UpdateID     INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    TaskID       INT NOT NULL,
    UserID       INT NOT NULL,
    [UpdateText] NVARCHAR(MAX) NOT NULL,
    NewStatus    NVARCHAR(20)  NULL,
    CreatedAt    DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_TaskUpdate_Task
        FOREIGN KEY (TaskID) REFERENCES dbo.Task(TaskID),
    CONSTRAINT FK_TaskUpdate_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID)
);

CREATE TABLE dbo.TaskComment (
    TaskCommentID INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    TaskID        INT NOT NULL,
    UserID        INT NOT NULL,
    Content       NVARCHAR(MAX) NOT NULL,
    IsEdited      BIT NOT NULL DEFAULT 0,
    CreatedAt     DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    UpdatedAt     DATETIME2 NULL,
    CONSTRAINT FK_TaskComment_Task
        FOREIGN KEY (TaskID) REFERENCES dbo.Task(TaskID) ON DELETE CASCADE,
    CONSTRAINT FK_TaskComment_User
        FOREIGN KEY (UserID) REFERENCES dbo.[User](UserID)
);
CREATE INDEX IX_TaskComment_TaskID ON dbo.TaskComment(TaskID);
CREATE INDEX IX_TaskComment_UserID ON dbo.TaskComment(UserID);

CREATE TABLE dbo.LifecycleStage (
    StageID      INT IDENTITY(1,1) NOT NULL PRIMARY KEY,

CREATE TABLE dbo.ProductHistory (
    HistoryID    INT IDENTITY(1,1) NOT NULL PRIMARY KEY,

CREATE TABLE dbo.Document (
    DocumentID      INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
    CompanyID       INT NOT NULL,
    FolderID        INT NOT NULL,
    ProjectID       INT NULL,
    Title           NVARCHAR(200) NOT NULL,
    Description     NVARCHAR(MAX) NULL,
    FileType        NVARCHAR(20)  NOT NULL,
    UploadedBy      INT NOT NULL,
    ApprovalStatus  NVARCHAR(20)  NOT NULL DEFAULT 'Pending',
    ApprovedBy      INT NULL,
    CreatedAt       DATETIME2 NOT NULL DEFAULT GETUTCDATE(),
    CONSTRAINT FK_Document_Company
        FOREIGN KEY (CompanyID) REFERENCES dbo.Company(CompanyID),
    CONSTRAINT FK_Document_Folder
        FOREIGN KEY (FolderID) REFERENCES dbo.Folder(FolderID),
    CONSTRAINT FK_Document_Project
        FOREIGN KEY (ProjectID) REFERENCES dbo.Project(ProjectID) ON DELETE SET NULL,
    CONSTRAINT FK_Document_UploadedBy
        FOREIGN KEY (UploadedBy) REFERENCES dbo.[User](UserID),
    CONSTRAINT FK_Document_ApprovedBy
        FOREIGN KEY (ApprovedBy) REFERENCES dbo.[User](UserID)
